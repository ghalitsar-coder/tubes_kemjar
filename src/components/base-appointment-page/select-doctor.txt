        function renderCalendar(date) {
            const month = date.getMonth();
            const year = date.getFullYear();
            
            // Update month/year display
            const monthNames = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
            currentMonthYearEl.textContent = `${monthNames[month]} ${year}`;
            
            // Get first day of month
            const firstDay = new Date(year, month, 1);
            const startingDay = firstDay.getDay(); // 0 = Sunday, 1 = Monday, etc.
            
            // Get last day of month
            const lastDay = new Date(year, month + 1, 0);
            const daysInMonth = lastDay.getDate();
            
            // Clear previous days
            calendarEl.innerHTML = '';
            
            // Add empty cells for days before the 1st of the month
            for (let i = 0; i < startingDay; i++) {
                const emptyCell = document.createElement('div');
                emptyCell.className = 'py-2';
                calendarEl.appendChild(emptyCell);
            }
            
            // Add day cells for the month
            const today = new Date();
            const currentDoctor = currentState.selectedDoctor;
            
            for (let day = 1; day <= daysInMonth; day++) {
                const dayDate = new Date(year, month, day);
                const weekday = dayDate.getDay();
                const dayName = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'][weekday];
                const dateString = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
                
                // Check if doctor is available on this day

                const isPast = dayDate < new Date(today.getFullYear(), today.getMonth(), today.getDate());
                
                const dayCell = document.createElement('div');
                dayCell.className = `py-2 text-center calendar-day rounded-md ${isAvailable && !isPast ? 'cursor-pointer hover:bg-indigo-50' : 'disabled-day cursor-not-allowed opacity-50'}`;
                
                // Highlight today
                if (day === today.getDate() && month === today.getMonth() && year === today.getFullYear()) {
                    dayCell.classList.add('border', 'border-indigo-200', 'bg-indigo-50');
                }
                
                // Highlight selected date
                if (currentState.selectedDate === dateString) {
                    dayCell.classList.add('bg-indigo-100', 'text-indigo-700');
                }
                
                dayCell.textContent = day;
                
                if (isAvailable && !isPast) {
                    dayCell.addEventListener('click', () => handleDateClick(dateString, day, month, year, dayName));
                }
                
                calendarEl.appendChild(dayCell);
            }
        }

        function renderDayNames() {
            const dayNames = ['S', 'M', 'T', 'W', 'T', 'F', 'S'];

                headerRow.appendChild(dayCell);
            });
            
            calendarEl.parentNode.insertBefore(headerRow, calendarEl);
        }

        function handleDateClick(dateString, day, month, year, dayName) {
            currentState.selectedDate = dateString;

            selectedDateDisplayEl.textContent = `${dayName}, ${monthNames[month]} ${day}, ${year}`;
            
            // Get time slots for this doctor and date

                    </p>
                `;
                confirmAppointmentBtn.disabled = true;
                currentState.selectedTime = null;
                return;

                const timeSlotEl = document.createElement('button');
                timeSlotEl.className = `w-full py-2 px-4 bg-white border rounded-md time-slot ${currentState.selectedTime === time ? 'border-indigo-500 bg-indigo-50 text-indigo-700' : 'border-gray-300 text-gray-700 hover:bg-gray-50'}`;
                timeSlotEl.textContent = time;

                        el.classList.remove('border-indigo-500', 'bg-indigo-50', 'text-indigo-700');
                        el.classList.add('border-gray-300', 'text-gray-700', 'hover:bg-gray-50');
                    });
                    timeSlotEl.classList.add('border-indigo-500', 'bg-indigo-50', 'text-indigo-700');
                    timeSlotEl.classList.remove('border-gray-300', 'text-gray-700', 'hover:bg-gray-50');
                    
                    currentState.selectedTime = time;
                    confirmAppointmentBtn.disabled = false;
                });
                timeSlotsContainerEl.appendChild(timeSlotEl);
            });
            
            // If previously selected time is still available, select it
            if (currentState.selectedTime && timeSlots.includes(currentState.selectedTime)) {
                const index = timeSlots.indexOf(currentState.selectedTime);
                timeSlotsContainerEl.children[index].click();
            } else {
                confirmAppointmentBtn.disabled = true;
                currentState.selectedTime = null;
            }
